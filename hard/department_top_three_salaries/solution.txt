with e as(
    select e.id, d.name as Department , e.name as Employee , e.salary as Salary ,
    dense_rank() over(partition by d.id order by e.salary desc) as rnk
    from 
    employee e
    join department d on e.departmentId = d.id
)
select Department , employee , salary from e
where rnk<=3;
